## 🔧 后续网站开发准备

### 数据结构建议

```typescript
// 需求分类枚举
enum RequirementCategory {
  NEW = 'new',           // 表格1: 新需求
  STATUS_UPDATE = 'update',  // 表格2: 状态更新
  CONFLICT = 'conflict',     // 表格3: 冲突
  UNCHANGED = 'unchanged'    // 无变化
}

// 需求数据结构
interface Requirement {
  id: string;
  title: string;
  category: RequirementCategory;
  assignee: string;
  priority: 'P0' | 'P1' | 'P2' | 'P3' | null;
  type: 'bug' | 'feature' | 'improvement' | 'other';
  userOriginalText: string;  // 用户原话
  
  // 表格2特有字段
  originalTitle?: string;    // 原需求标题
  mergedTitle?: string;      // 合并后需求
  statusChange?: {           // 状态变更
    from: string;
    to: string;
  };
  
  // 表格3特有字段
  conflictDescription?: string;  // 冲突描述
  
  // 元数据
  rawContext: {
    source: 'feishu' | 'wechat';
    rawContent: string;
    rawImages: Array<{
      type: string;
      aeskey?: string;
      url?: string;
    }>;
    conversationContext?: {
      before: Array<{content: string; timestamp: string}>;
      current: {content: string; timestamp: string};
      after: Array<{content: string; timestamp: string}>;
    };
  };
}
```
